32  */
33 void usart_init(void)
34 {
35  uint32_t ubrr = (CLK_SPEED/16UL)/BAUD-1; // calculates value for Baud Rate register
36  UBRR0H = (unsigned char)(ubrr>>8); // based on 9600 baud rate and 16MHz clock speed.
37  UBRR0L = (unsigned char)ubrr; // assigns this vallue to the baud rate register.
38  UCSR0B = (1<<RXEN0)|(1<<TXEN0); // enable transmit and receive
39  UCSR0C = (0<<USBS0)|(3<<UCSZ00); // configure for 1 stop bit, with an 8 character data packet.
40  UCSR0A &= ~(1<<U2X0); // ensure double uart speed is cleared.
41 }

44 /** \brief This Function prompts the UART to transmit a byte.
45  */
46 void usart_send( uint8_t data )
47 {
48  while ( !( UCSR0A & (1<<UDRE0)) ); // Wait for empty transmit buffer
49  UDR0 = data; // Put data into buffer, sends the data
50  return;
51 }
