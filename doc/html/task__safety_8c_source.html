42 /** \brief This task function for the safety task.
43  * \details monitors the the motor diag_enable A/B pins, and the
44  * current, and shuts the motors off if any problems are detected.
45  */
46 void task_safety(void* pvParameters){
47  uint8_t default_safety_prio = uxTaskPriorityGet(NULL);
48  portTickType xLastWakeTime;
49  xLastWakeTime = xTaskGetTickCount();
50  uint8_t safety_error_SHARED = 0;
51  uint16_t current_val_M1 = 0;
52  uint16_t current_val_M2 = 0;
53  const char *over_current_m1 = "Error: Excess current in Motor 1!";
54  const char *over_current_m2 = "Error: Excess current in Motor 2!";
55  const char *fault_m1 = "Error: Fault on Motor 1!";
56  const char *fault_m2 = "Effor: Fault on Motor 2!";
57 
58  while(1)
59  {
60  // Check for over current in motor 1.
61  current_val_M1 = adc_read(ADC_CURRENT_M1);
62  if(current_val_M1>MAX_CURRENT_M1)
63  {
64  // Critical section unnecessary for accessing the shared var,
65  // since the safety task has the top priority :D
66  // (also no ISR touches this var)
67  safety_error_SHARED = 1; //Flag an error!
68  xQueueSend(comms_queue,&over_current_m1,0);
69  }
70 
71  // Check for over current in motor 2.
72  current_val_M2 = adc_read(ADC_CURRENT_M2);
73  if(current_val_M2>MAX_CURRENT_M2)
74  {
75  safety_error_SHARED = 1;//Flag an error!
76  xQueueSend(comms_queue,&over_current_m2,0);
77  }
78 
79  // Check for fault on motor 1.
80  if( !(PINC & (1<<EN_AB_M1)) )
81  {
82  safety_error_SHARED = 1;//Flag an error!
83  xQueueSend(comms_queue,&fault_m1,0);
84  }
85 
86  // Check for fault on motor 2.
87  if( !(PINC & (1<<EN_AB_M2)) )
88  {
89  safety_error_SHARED = 1;//Flag an error!
90  xQueueSend(comms_queue,&fault_m2,0);
91  }
92 
93  vTaskDelayUntil(&xLastWakeTime, 100/portTICK_RATE_MS);
94  }

96 
97 }
