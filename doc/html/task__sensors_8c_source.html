56 void adc_init(void){
57  // Set the "ADC Enable" bit within the ADC Status Register A,
58  // and set the ADC clock prescaler to divide by 32. Enable
59  // interrupt on conversion complete.
60  ADCSRA = (1<<ADIE) | (1<<ADEN) | (1<<ADPS2) | (1<<ADPS0);
61  ADMUX |= 1<<REFS0;// Set the analog reference voltage to AVCC.
62 
63 }

66 /** \brief This Function reads a byte from the selected ADC channel.
67  * @param adc_channel Takes one of the #define macros from task_sensors.h as a
68  * parameter, for the purpose of selecting one of the ADC channels.
69  * @return Returns and unsigned 16bit data value.
70  */
71 uint16_t adc_read(uint8_t adc_channel){
72  /// Use a mutex to control access to the ADC.
73  xSemaphoreTake(adc_mutex_semaphore, portMAX_DELAY);
74 
75  // Clear the ADC Multiplexor Selection bits, and the set them to
76  // the appropriate ADC channel.
77  ADMUX &= ~( (1<<MUX2)|(1<<MUX1)|(1<<MUX0) );
78  ADMUX |= adc_channel;
79 
80  // Set the "ADC start conversion" bit, initiating a conversion.
81  ADCSRA |= 1<<ADSC;
82  const uint16_t maxWaitCycles = 60000;
83  uint16_t counter = 0;
84  while(ADCSRA & (1<<ADSC) && maxWaitCycles > counter++);
85  uint16_t temp = ADC;
86 
87  xSemaphoreGive(adc_mutex_semaphore);
88  return temp;
89 
90 }

152 /** \brief This is the task function for monitoring the button and joysticks.
153  * \details This function monitors the "Target Set" button, and when it is pressed,
154  * updates shared variables "motor1_power_SHARED", and "motor2_power_SHARED" with adc_channel
155  * readings. If the button is NOT pressed, the motor power shareds are set to the effective
156  * "zero power" value of 512.(1024 adc values corresponding to full forward(1023), and
157  * full reverse(0);
158  */
159 void task_sensors(void* pvParameters){
160  uint8_t default_sensor_prio = uxTaskPriorityGet(NULL);
161  portTickType xLastWakeTime;
162  xLastWakeTime = xTaskGetTickCount();
163  adc_init();
164  uint16_t joystick_y;
165  uint16_t joystick_x;
166  button_init();
167 
168  while(1)
169  {
170  if(button_pressed()){
171  joystick_y = adc_read(ADC_JOYSTICK_Y);
172  joystick_x = adc_read(ADC_JOYSTICK_X);
173  vTaskPrioritySet(NULL, configMAX_PRIORITIES - 1);
174  motor1_power_SHARED = joystick_y;
175  motor2_power_SHARED = joystick_x;
176  vTaskPrioritySet(NULL, default_sensor_prio);
177  }
178  else{
179  vTaskPrioritySet(NULL, configMAX_PRIORITIES - 1);
180  motor1_power_SHARED = 512;
181  motor2_power_SHARED = 512;
182  vTaskPrioritySet(NULL, default_sensor_prio);
183  }
184  vTaskDelayUntil(&xLastWakeTime, 100/portTICK_RATE_MS);
185  }

187 
188 }
