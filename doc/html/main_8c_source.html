112 /** \brief This ISR updates the motors position when a pin change interrupt has occured
113  * as a result of a change in the encoder output waveforms.
114  */
115 ISR(PCINT0_vect){
116  //previous state of Motor 1 is saved via static var
117  static uint8_t previous_state_M1;
118 
119  //previous state of Motor 2 is saved via static var
120  static uint8_t previous_state_M2;
121 
122  //get new encoder state
123  uint8_t state_M1 = ( (1<<ENC_A_M1)|(1<<ENC_B_M1) ) & PINA;
124  uint8_t state_M2 = ( (1<<ENC_A_M2)|(1<<ENC_B_M2) ) & PINA;
125 
126  state_M2 = (state_M2>>2); // Motor 2's encoders are on PORT A, bits 2 and 3. This
127  // line of code maps their possible states to unsigned
128  // integer values 0 through 3.
129 
130  //check encoder state sequence
131  const uint8_t sequence[6]={3,2,0,1,3,2};
132 
133  for (uint8_t i=1; i<5;i++)
134  {
135  if (state_M1==sequence[i])
136  {
137  //check for ccw, write to dir
138  if (previous_state_M1==sequence[i-1])
139  {
140  position_M1_SHARED = position_M1_SHARED - 1;
141  break;
142  }
143  //check for cw, write to dir
144  else if(previous_state_M1==sequence[i+1])
145  {
146  position_M1_SHARED = position_M1_SHARED + 1;
147  break;
148  }
149  //if same, pin change likely occured on M2
150  else if (previous_state_M1==sequence[i])
151  {
152  position_M1_SHARED = position_M1_SHARED;
153  break;
154  }
155  //check for error, write to error
156  else
157  {
158  error_M1_SHARED++;
159  break;
160  }
161  }
162  }
163  for (uint8_t i=1; i<5;i++)
164  {
165  if (state_M2==sequence[i])
166  {
167  //check for ccw, write to dir
168  if (previous_state_M2==sequence[i-1])
169  {
170  position_M2_SHARED = position_M2_SHARED - 1;
171  break;
172  }
173  //check for cw, write to dir
174  else if(previous_state_M2==sequence[i+1])
175  {
176  position_M2_SHARED = position_M2_SHARED + 1;
177  break;
178  }
179  //if same, pin change likely occured on M2
180  else if (previous_state_M2==sequence[i])
181  {
182  position_M2_SHARED = position_M2_SHARED;
183  break;
184  }
185  //check for error, write to error
186  else
187  {
188  error_M2_SHARED++;
189  break;
190  }
191  }
192  }
193  //save previous state
194  previous_state_M1 = state_M1;
195  previous_state_M2 = state_M2;
196 }
