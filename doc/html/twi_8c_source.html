65 /** \brief This Function prompts the AVR I2C device to read one or more bytes.
66  * \details This Function prompts the I2C device to read data from the I2C
67  * data lines in master receiver mode. If the argument "lastbit"
68  * is 0, then after receiving the last byte of information, the
69  * acknowledge bit is not pulled low - this indicates the end of
70  * the transmission. If "lastbit" is set to 1, the acknowledge bit
71  * will be sent, and the I2C device will continue to read data.
72  *
73  * @param lastbit This tells the function whether or not pull the i2c/twi data lines
74  * low for an "ACK" condition.
75  * @return Returns the single byte of recieved data.
76  */
77 uint8_t twi_read(uint8_t lastbit)
78 {
79  if(lastbit == 0)
80  {
81  TWCR = ( (1<<TWINT) | (1<<TWEN) );
82  }
83  else
84  {
85  TWCR = ( (1<<TWINT) | (1<<TWEA) | (1<<TWEN) );
86  }
87 
88  while( ((1<<TWINT) & TWCR) == 0 )
89  {
90  __asm__("nop");
91  }
92 
93  return TWDR;
94 }
