91 /** \brief This task function reads sensor data, and calculates the desired mirror angle
92  * \details This function wakes up every four minutes, determines time based on RTC
93  * sensor, determines heading with the magnetometer, calculates the angle of the sun,
94  * calculates the desired angle of the mirror array, commands that the array track
95  * the calculated angle, and blocks itself for another four minutes.
96  */
97 void task_orient(void* pvParameters){
98  uint8_t default_orient_prio = uxTaskPriorityGet(NULL);
99  portTickType xLastWakeTime;
100  xLastWakeTime = xTaskGetTickCount();
101  HMC5883_init();
102  while(1)
103  {
104  /// The priority is set to max, since HMC5883_read accesses a shared variable.
105  vTaskPrioritySet(NULL, configMAX_PRIORITIES - 1);
106  HMC5883_read();
107  vTaskPrioritySet(NULL, default_orient_prio);
108  position_cmd_M1_SHARED = 0; // To be changed as soon as the orientation algorithm
109  position_cmd_M2_SHARED = 0; // is implemented.
110  vTaskDelayUntil(&xLastWakeTime, 600000/portTICK_RATE_MS);
111  }
112 }
